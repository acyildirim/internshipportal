public async Task<IActionResult> Apply(int? id)
        {
            if (id == null)
            {
                return NotFound();
            }

            var job = await _context.Job
                .FirstOrDefaultAsync(m => m.Id == id);
            if (job == null)
            {
                return NotFound();
            }
            _context.Add(new JobApplicant
            {
                JobID = id.Value,
                UserID = User.Identity.Name

        });
            await _context.SaveChangesAsync();
            return View(job);
        }
